Topics : Dp , Bit manipulation

class Solution {
    public int subarrayBitwiseORs(int[] arr) {
        Set<Integer> prev = new HashSet<>();
        Set<Integer> curr = new HashSet<>();
        Set<Integer> ans = new HashSet<>();
        int n = arr.length;
        for(int i = 0 ; i < n ; i++){
            curr.add(arr[i]);
            for(int p : prev){
                curr.add(arr[i] | p);
            }
            for(int c : curr){
                ans.add(c);
            }
            ans.addAll(curr);
            prev.clear();
            prev.addAll(curr);
            curr.clear();
        }
        return ans.size();
    }
}

Time Complexity: O(NlogM);
Auxiliary Space: O(NlogM);
