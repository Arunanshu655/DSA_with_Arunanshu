Topics: Prefix-Sum, Math

class Solution {
    public int cntSubarrays(int[] arr, int k) {
        // code here
        int sum = 0 , ans = 0;
        HashMap<Integer,Integer> map = new HashMap<>();
        map.put(0,1);
        for(int a : arr){
            sum += a;
            if(map.containsKey(sum-k)) ans += map.get(sum-k);
            map.put(sum,map.getOrDefault(sum,0)+1);
        }
        return ans;
    }
}

Time Complexity: O(n)
Auxiliary Space: O(n)
